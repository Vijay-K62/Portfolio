# Terraform configuration for EKS Cluster
module "vpc" {
  source = "./modules/vpc"
  vpc_cidr             = var.vpc_cidr
  public_subnet_cidrs  = var.public_subnet_cidrs
  private_subnet_cidrs = var.private_subnet_cidrs
  environment          = var.environment
}

module "eks" {
  source = "./modules/eks"
  cluster_name    = "${var.environment}-eks-cluster"
  vpc_id          = module.vpc.vpc_id
  private_subnets = module.vpc.private_subnet_ids
  node_groups = {
    general = {
      desired_size = 2
      max_size     = 4
      min_size     = 1
      instance_types = ["t3.medium"]
    }
  }
}
variable "environment" {
  type        = string
  description = "Deployment stage name"
  default     = "production"
}

variable "vpc_cidr" {
  type    = string
  default = "10.0.0.0/16"
}

variable "public_subnet_cidrs" {
  type    = list(string)
  default = ["10.0.1.0/24", "10.0.2.0/24"]
}

variable "private_subnet_cidrs" {
  type    = list(string)
  default = ["10.0.3.0/24", "10.0.4.0/24"]
}
terraform {
  required_version = ">= 1.6.0"
  
  backend "s3" {
    bucket         = "saivijay-tf-state"
    key            = "state/production.tfstate"
    region         = "ap-south-1"
    encrypt        = true
    dynamodb_table = "saivijay-tf-lock-table"
  }
}
output "vpc_id" {
  value       = module.vpc.vpc_id
  description = "The ID of the generated VPC"
}

output "eks_cluster_endpoint" {
  value       = module.eks.cluster_endpoint
  description = "Kubernetes control plane API endpoint"
}

output "eks_cluster_name" {
  value       = module.eks.cluster_name
}
